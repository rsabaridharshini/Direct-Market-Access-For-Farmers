    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        // TODO add your handling code here:
        String name = usernamefield.getText();
    String pass = new String(passwordfield.getPassword());
    String category = "";
    if (fb.isSelected()) {
        category = "farmer";
    } else if (cb.isSelected()) {
        category = "consumer";
    }
try {
    if (name.isEmpty() || pass.isEmpty()||category.isEmpty()) {
        JOptionPane.showMessageDialog(this, "Please fill in all fields", "Error", JOptionPane.ERROR_MESSAGE);
        return;
    }
    
               Class.forName("oracle.jdbc.OracleDriver");
        
        // Establish connection to the Oracle database
        Connection con = DriverManager.getConnection("jdbc:oracle:thin:@localhost:1521:XE", "system", "admin");
        
        // Prepare the SQL SELECT statement
        // Prepare the SQL SELECT statement
         String sql = "SELECT category,username,password FROM flogin WHERE category=? AND username = ? AND password = ?";

        PreparedStatement pst = con.prepareStatement(sql);
        
        // Set values for the parameters in the SELECT statement
        pst.setString(1, category);
        pst.setString(2, name);
        pst.setString(3, pass);
        
        // Execute the SELECT statement
        ResultSet rs = pst.executeQuery(); 
            if (rs.next()) {
                JOptionPane.showMessageDialog(this, "Login successful!");
                if(category=="consumer")
                {
                fdashboard1 o=new fdashboard1();
                o.setVisible(true);
                dispose();
                }
                else
                {
                fdashboard2 o=new fdashboard2();
                o.setVisible(true);
                dispose();
                }
            } else {
                JOptionPane.showMessageDialog(this, "Invalid login. Please try again.");
            }
            rs.close();
        pst.close();
        con.close();
        
    } catch (ClassNotFoundException ex) {
        ex.printStackTrace();
        JOptionPane.showMessageDialog(this, "Error: " + ex.getMessage(), "Error", JOptionPane.ERROR_MESSAGE);
    } catch (SQLException ex) {
        ex.printStackTrace();
        JOptionPane.showMessageDialog(this, "Error: " + ex.getMessage(), "Error", JOptionPane.ERROR_MESSAGE);
    }
        
    }
