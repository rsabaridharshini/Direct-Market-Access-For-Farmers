private void loadTableData() {
        DefaultTableModel model = (DefaultTableModel) table.getModel();
        model.setRowCount(0);
        
        try {
            Class.forName("oracle.jdbc.driver.OracleDriver");
            // Establish connection to the database
            Connection conn = DriverManager.getConnection("jdbc:oracle:thin:@localhost:1521:xe", "system", "admin");

            // Query to fetch data from fimage table
            String query = "SELECT id, quantity, address FROM forder";
            PreparedStatement stmt = conn.prepareStatement(query);
            ResultSet rs = stmt.executeQuery();

            // Loop through the result set
            while (rs.next()) {
                int id = rs.getInt("id");
                String quantity = rs.getString("quantity");
                String address = rs.getString("address");
                model.addRow(new Object[]{id,quantity,address});
                // Convert image blob to ImageIcon for display in JTable
            }

            // Close resources
            rs.close();
            stmt.close();
            conn.close();
        } catch (Exception e) {
            e.printStackTrace();
        }
}
